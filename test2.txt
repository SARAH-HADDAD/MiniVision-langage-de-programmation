import numpy as np

# Charger l'image en tant qu'array numpy
img = np.array([[255, 100, 50],
                [200, 150, 75],
                [100, 50, 25]])

# Parcourir chaque pixel de l'image et inverser nÃ©gativement-positivement
for i in range(img.shape[0]):
    for j in range(img.shape[1]):
        img[i, j] = 255 - img[i, j]

print(img)
